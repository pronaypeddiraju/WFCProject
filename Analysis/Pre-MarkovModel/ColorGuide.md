	-	WHITE (255,255,255): 			Empty
	-	BLACK (0,0,0): 					Wall	
	-	PURE BLUE (0, 0, 255):			Glass Wall
	-	PURE RED (255, 0, 0):			Wet Wall

#Interiors#

	-	LIGHT BROWN (255, 128, 0): 		Tables
	-	PURE YELLOW (255, 255, 0):		Chairs
	- 	DARK BROWN (128, 64, 32) :		Shelf
